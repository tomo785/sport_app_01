# 核心数据模型设计

## 一、数据模型概览

基于关系型数据库 MySQL 设计，采用表结构存储数据。

```
Database: sport_app
│
├── t_user                     # 用户表
├── t_workout_record           # 运动记录表
├── t_workout_trajectory       # 运动轨迹点表
├── t_workout_goal             # 运动目标表
├── t_user_daily_stats         # 用户每日统计表
└── t_sms_code                 # 短信验证码表
```

---

## 二、用户数据模型

### 1. t_user - 用户表

**表名**: `t_user`

**字段说明**:

| 字段名 | 类型 | 长度 | 允许NULL | 默认值 | 说明 |
|--------|------|------|----------|--------|------|
| id | BIGINT | - | NO | AUTO | 用户ID（主键） |
| phone | VARCHAR | 20 | YES | NULL | 手机号 |
| password | VARCHAR | 100 | YES | NULL | 密码（BCrypt加密） |
| openid | VARCHAR | 64 | YES | NULL | 微信OpenID |
| unionid | VARCHAR | 64 | YES | NULL | 微信UnionID |
| nickname | VARCHAR | 50 | YES | NULL | 昵称 |
| avatar | VARCHAR | 255 | YES | NULL | 头像URL |
| gender | TINYINT | - | YES | 0 | 性别：0-未知 1-男 2-女 |
| age | INT | - | YES | NULL | 年龄 |
| height | INT | - | YES | NULL | 身高（cm） |
| weight | DECIMAL | 5,2 | YES | NULL | 体重（kg） |
| login_type | TINYINT | - | YES | 1 | 登录方式：1-手机号 2-微信 |
| status | TINYINT | - | NO | 1 | 状态：0-禁用 1-正常 |
| create_time | DATETIME | - | NO | NOW() | 创建时间 |
| update_time | DATETIME | - | NO | NOW() | 更新时间 |
| delete_flag | TINYINT | - | NO | 0 | 逻辑删除：0-未删除 1-已删除 |

**索引**:
```sql
PRIMARY KEY (id),
UNIQUE KEY uk_phone (phone),
UNIQUE KEY uk_openid (openid),
KEY idx_create_time (create_time)
```

**实体类**:
```java
@Data
@TableName("t_user")
public class User {
    @TableId(type = IdType.AUTO)
    private Long id;

    private String phone;
    private String password;
    private String openid;
    private String unionid;
    private String nickname;
    private String avatar;
    private Integer gender;    // 0-未知 1-男 2-女
    private Integer age;
    private Integer height;     // cm
    private BigDecimal weight; // kg
    private Integer loginType;  // 1-手机号 2-微信
    private Integer status;     // 0-禁用 1-正常

    @TableField(fill = FieldFill.INSERT)
    private LocalDateTime createTime;

    @TableField(fill = FieldFill.INSERT_UPDATE)
    private LocalDateTime updateTime;

    @TableLogic
    private Integer deleteFlag;
}
```

**表名**: `t_workout_record`

**字段说明**:

| 字段名 | 类型 | 长度 | 允许NULL | 默认值 | 说明 |
|--------|------|------|----------|--------|------|
| id | BIGINT | - | NO | AUTO | 运动记录ID（主键） |
| user_id | BIGINT | - | NO | - | 用户ID（外键） |
| type | TINYINT | - | NO | - | 运动类型：1-跑步 2-健走 3-骑行 |
| status | TINYINT | - | NO | - | 状态：1-进行中 2-暂停 3-已完成 4-已放弃 |
| start_time | DATETIME | - | NO | - | 开始时间 |
| end_time | DATETIME | - | YES | NULL | 结束时间 |
| duration | INT | - | YES | 0 | 运动时长（秒） |
| distance | INT | - | YES | 0 | 总距离（米） |
| calories | INT | - | YES | 0 | 总卡路里（千卡） |
| avg_speed | DECIMAL | 8,2 | YES | NULL | 平均速度（米/秒） |
| max_speed | DECIMAL | 8,2 | YES | NULL | 最大速度（米/秒） |
| min_altitude | INT | - | YES | NULL | 最低海拔（米） |
| max_altitude | INT | - | YES | NULL | 最高海拔（米） |
| total_ascent | INT | - | YES | 0 | 总爬升（米） |
| total_descent | INT | - | YES | 0 | 总下降（米） |
| trajectory_count | INT | - | YES | 0 | 轨迹点总数 |
| create_time | DATETIME | - | NO | NOW() | 创建时间 |
| update_time | DATETIME | - | NO | NOW() | 更新时间 |

**索引**:
```sql
PRIMARY KEY (id),
KEY idx_user_id (user_id),
KEY idx_user_time (user_id, start_time DESC),
KEY idx_type (type),
KEY idx_status (status)
```

**实体类**:
```java
@Data
@TableName("t_workout_record")
public class WorkoutRecord {
    @TableId(type = IdType.AUTO)
    private Long id;

    private Long userId;
    private Integer type;      // 1-跑步 2-健走 3-骑行
    private Integer status;    // 1-进行中 2-暂停 3-已完成 4-已放弃

    private LocalDateTime startTime;
    private LocalDateTime endTime;

    private Integer duration;  // 秒
    private Integer distance;  // 米
    private Integer calories;  // 千卡

    private BigDecimal avgSpeed;   // 米/秒
    private BigDecimal maxSpeed;   // 米/秒

    private Integer minAltitude;    // 米
    private Integer maxAltitude;    // 米
    private Integer totalAscent;    // 米
    private Integer totalDescent;  // 米

    private Integer trajectoryCount;

    @TableField(fill = FieldFill.INSERT)
    private LocalDateTime createTime;

    @TableField(fill = FieldFill.INSERT_UPDATE)
    private LocalDateTime updateTime;
}
```

## 九、数据库初始化脚本

**schema.sql**:
```sql
-- 创建数据库
CREATE DATABASE IF NOT EXISTS sport_app DEFAULT CHARSET utf8mb4 COLLATE utf8mb4_unicode_ci;

USE sport_app;

-- 用户表
CREATE TABLE t_user (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '用户ID',
    phone VARCHAR(20) UNIQUE COMMENT '手机号',
    password VARCHAR(100) COMMENT '密码',
    openid VARCHAR(64) UNIQUE COMMENT '微信OpenID',
    unionid VARCHAR(64) COMMENT '微信UnionID',
    nickname VARCHAR(50) COMMENT '昵称',
    avatar VARCHAR(255) COMMENT '头像URL',
    gender TINYINT DEFAULT 0 COMMENT '性别：0-未知 1-男 2-女',
    age INT COMMENT '年龄',
    height INT COMMENT '身高（cm）',
    weight DECIMAL(5,2) COMMENT '体重（kg）',
    login_type TINYINT DEFAULT 1 COMMENT '登录方式：1-手机号 2-微信',
    status TINYINT DEFAULT 1 COMMENT '状态：0-禁用 1-正常',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    delete_flag TINYINT DEFAULT 0 COMMENT '逻辑删除：0-未删除 1-已删除',
    INDEX idx_create_time (create_time)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- 运动记录表
CREATE TABLE t_workout_record (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '运动记录ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    type TINYINT NOT NULL COMMENT '运动类型：1-跑步 2-健走 3-骑行',
    status TINYINT NOT NULL COMMENT '状态：1-进行中 2-暂停 3-已完成 4-已放弃',
    start_time DATETIME NOT NULL COMMENT '开始时间',
    end_time DATETIME COMMENT '结束时间',
    duration INT DEFAULT 0 COMMENT '运动时长（秒）',
    distance INT DEFAULT 0 COMMENT '总距离（米）',
    calories INT DEFAULT 0 COMMENT '总卡路里（千卡）',
    avg_speed DECIMAL(8,2) COMMENT '平均速度（米/秒）',
    max_speed DECIMAL(8,2) COMMENT '最大速度（米/秒）',
    min_altitude INT COMMENT '最低海拔（米）',
    max_altitude INT COMMENT '最高海拔（米）',
    total_ascent INT DEFAULT 0 COMMENT '总爬升（米）',
    total_descent INT DEFAULT 0 COMMENT '总下降（米）',
    trajectory_count INT DEFAULT 0 COMMENT '轨迹点总数',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX idx_user_id (user_id),
    INDEX idx_user_time (user_id, start_time DESC),
    INDEX idx_type (type),
    INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='运动记录表';

-- 运动轨迹点表
CREATE TABLE t_workout_trajectory (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '轨迹点ID',
    workout_id BIGINT NOT NULL COMMENT '运动记录ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    latitude DECIMAL(10,7) NOT NULL COMMENT '纬度',
    longitude DECIMAL(10,7) NOT NULL COMMENT '经度',
    altitude INT COMMENT '海拔（米）',
    accuracy DECIMAL(6,2) COMMENT '定位精度（米）',
    speed DECIMAL(8,2) COMMENT '速度（米/秒）',
    heading DECIMAL(6,2) COMMENT '方向（度，0-360）',
    timestamp DATETIME NOT NULL COMMENT '记录时间',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    INDEX idx_workout_id (workout_id),
    INDEX idx_user_id (user_id),
    INDEX idx_timestamp (timestamp)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='运动轨迹点表';

-- 运动目标表
CREATE TABLE t_workout_goal (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '目标ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    type TINYINT NOT NULL COMMENT '目标类型：1-距离 2-时长 3-卡路里 4-次数',
    workout_type TINYINT COMMENT '运动类型：1-跑步 2-健走 3-骑行',
    target_value INT NOT NULL COMMENT '目标值',
    period TINYINT NOT NULL COMMENT '周期：1-每日 2-每周 3-每月',
    current_value INT DEFAULT 0 COMMENT '当前完成值',
    progress DECIMAL(5,2) DEFAULT 0.00 COMMENT '完成百分比（0-100）',
    is_completed TINYINT DEFAULT 0 COMMENT '是否完成：0-未完成 1-已完成',
    start_date DATE NOT NULL COMMENT '开始日期',
    end_date DATE COMMENT '结束日期',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    delete_flag TINYINT DEFAULT 0 COMMENT '逻辑删除：0-未删除 1-已删除',
    INDEX idx_user_id (user_id),
    INDEX idx_period (period),
    INDEX idx_status (is_completed, delete_flag)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='运动目标表';

-- 用户每日统计表
CREATE TABLE t_user_daily_stats (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '统计ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    stat_date DATE NOT NULL COMMENT '统计日期',
```
